create table Party(
	partyId SERIAL Primary Key,
	firstName varchar(20) Not Null,
	lastname varchar(20) Not Null,
	address text,
	city varchar(20),
	zip varchar(10),
	state text,
	country varchar(20),
	phone varchar(10)
);

create table UserLogin(
	userLoginId varchar(50) Primary Key,
	password varchar(100) Not Null,
	partyId Integer References Party(partyId) on delete cascade
);
